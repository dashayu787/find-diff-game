function create() { const scene = this; const sw = this.scale.width; const sh = this.scale.height; const scaleX = sw / BASE_WIDTH; const scaleY = sh / BASE_HEIGHT; hitRadiusWorld = HIT_RADIUS_BASE * scaleX; // 上图 const topH = sh * IMAGE_H_RATIO; topImage = this.add.image(sw / 2, topH / 2, 'topImage') .setDisplaySize(sw, topH); topY0 = topImage.y - topImage.displayHeight / 2; // 绿色条（与上图左右齐边，紧贴上图底部） const barH = BAR_H_BASE * scaleY; const padding = PADDING_BASE * scaleX; const barTop = topY0 + topImage.displayHeight; const barCenter = barTop + barH / 2; this.add.rectangle(sw / 2, barCenter, sw, barH, 0x009739).setOrigin(0.5); // —— 一行文字：左(倒计时) — 中(标题) — 右(得分) —— const styleSmall = { fontFamily: 'Arial', fontSize: '14px', color: '#FFD700', fontStyle: 'bold' }; const styleTitle = { fontFamily: 'Arial', fontSize: '18px', color: '#FFD700', fontStyle: 'bold' }; timerText = this.add.text(padding, barCenter, 'TMP: 60', styleSmall) .setOrigin(0, 0.5); // 左对齐 + 垂直居中 scoreText = this.add.text(sw - padding, barCenter, 'PTS: 0', styleSmall) .setOrigin(1, 0.5); // 右对齐 + 垂直居中 titleText = this.add.text(sw / 2, barCenter, '97 Diferenças', styleTitle) .setOrigin(0.5, 0.5); // 居中 // 让标题始终位于“倒计时”和“得分”的中点（保持一行对齐） function layoutHeader() { const y = barCenter; timerText.setY(y); scoreText.setY(y); const leftEdge = timerText.x + timerText.width; // 左侧文字的右边界 const rightEdge = scoreText.x - scoreText.width; // 右侧文字的左边界 const mid = (leftEdge + rightEdge) / 2; // 两者中点 titleText.setPosition(mid, y); // 标题放在中点 } layoutHeader(); // 下图（与绿色条下边紧贴） const bottomH = sh * IMAGE_H_RATIO; const bottomTop = barTop + barH; bottomImage = this.add.image(sw / 2, bottomTop + bottomH / 2, 'bottomImage') .setDisplaySize(sw, bottomH); bottomY0 = bottomImage.y - bottomImage.displayHeight / 2; // 点击命中 this.input.on('pointerdown', (pointer) => { if (gameOver) return; const mx = pointer.x, my = pointer.y; for (let i = 0; i < points.length; i++) { if (foundSet.has(i)) continue; const { u, v } = points[i]; const px = u * sw; const pyTop = topY0 + v * topImage.displayHeight; const pyBottom = bottomY0 + v * bottomImage.displayHeight; const hit = (Phaser.Math.Distance.Between(mx, my, px, pyTop) <= hitRadiusWorld) || (Phaser.Math.Distance.Between(mx, my, px, pyBottom) <= hitRadiusWorld); if (hit) { foundSet.add(i); score++; scoreText.setText('PTS: ' + score); drawHit.call(scene, px, pyTop); drawHit.call(scene, px, pyBottom); layoutHeader(); // 分数变化后重新排版（保持一行） if (score >= points.length) endGame.call(scene, true); return; } } }); // 计时 timerEvt = this.time.addEvent({ delay: 1000, loop: true, callback: () => { if (gameOver) return; timeLeft--; timerText.setText('TMP: ' + timeLeft); timerText.setColor(timeLeft <= 10 ? '#ff0000' : '#FFD700'); layoutHeader(); // 文本变更后重新排版（保持一行） if (timeLeft <= 0) endGame.call(scene, false); } }); }
